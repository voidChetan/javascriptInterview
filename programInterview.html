Remove Duplicates from array 
                    let cityArray = ["Nagpur","Pune","Nagpur","Mumbai","Mumbai","Nagpur","Pune",];
                    function getUnique(arr) {
                      return [... new Set(arr)];
                    }
                    console.log(getUnique(cityArray));
                    Output:  ['Nagpur', 'Pune', 'Mumbai']
  
                    function getUniqueNativeType(arr) {
                        return arr.filter(function(item,pos){
                          return arr.indexOf(item) ==pos;
                        })
                    }
  
                    console.log(getUniqueNativeType(cityArray));
                    Output:  ['Nagpur', 'Pune', 'Mumbai']
